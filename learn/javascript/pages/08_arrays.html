آرایه‌ها [1]

در تمام زبان‌های برنامه‌نویسی مانند C++، Java و PHP مفهوم آرایه‌ها وجود دارد و از آن‌ها استفاده می‌شود. از همان مفاهیم در JS نیز استفاده می‌شود. آرایه‌ها لیستی از مقادیر هستند. قبلاً متغیرها را دیده‌ایم، فرض کنیم متغیری داریم که در آن بتوانیم لیست مقادیر را ذخیره سازی کنیم.

راه‌های مختلفی برای ساخت آرایه در JS وجود دارد؛ یکی از آن راه‌ها استفاده از کلاس[2] Array می‌باشد، برای درک بهتر به تکه کد زیر دقت کنید:

var animals = new Array(); 
animals[0]="cat"; 
animals[1]="dog"; 
animals[2]="mouse"

در کد بالا، با استفاده از تکه کد new Array() یک شی[23] از کلاس Array ایجاد شده است. در ادامه برنامه‌ برای خانه‌های 0، 1 و 2 مقادیری ذخیره شده است.

در برنامه‌نویسی به شماره خانه‌های هر آیتم در آرایه به اصطلاح ایندکس می‌گویند. برای مثال در کد بالا، مقداری که در ایندکس صفر آرایه animals ذخیره شده است، مقدال cat است.

💡 ایندکس‌های آرایه در JS از شماره 0 آغاز می‌شوند.

روش دیگر تعریف آرایه به شرح زیر است:

var animals = new Array("cat", "dog", "mouse");

تکه کد بالا؛ با کد قبلی تفاوتی از نظر کارکرد ندارد و فقط مقادیر از طریق سازنده [3] کلاس Array به آن انتساب داده شده است.

روش دیگر ایجاد آرایه: می‌توان بدون استفاده از کلاس Array نسبت به ایجاد آرایه اقدام کرد، برای این کار کافیست مقادیری که می‌خواهیم را بین دو [] قرار دهیم. برای مثال:

var animals = ["cat", "dog", "mouse"];

نحوه دسترسی به مقادیر داخل آرایه

برای دسترسی به مقادیر داخل آرایه کافیست به شکل زیر عمل کنیم:

نام_آرایه[index];

برای مثال:

var animals = ["cat", "dog", "mouse"];

alert(animals[0]); // alert("cat");
alert(animals[2]); // alert("mouse");

توابع داخلی Array

کلاس Array دارای برخی خصوصیت[4] و تابع/متد داخلی است که در اینجا شرح آن‌ها آمده است:

نام متد یا خصوصیت نوع مثال شرح
length خصوصیت animals.length این خصوصیت تعداد عناصر یک آرایه را باز می‌گرداند.
push متد animals.push(value) یک آیتم به انتهای آرایه اضافه می‌کند.
pop متد animals.push() یک آیتم از انتهای آرایه حذف و آن مقدار را باز می‌گرداند.
reverse متد animals.reverse() یک آرایه را معکوس می‌کند.
sort متد animals.sort() یک آرایه را مرتب می‌کند.

برای درک بهتر دستورات بالا؛ در تکه کد زیر برنامه‌ای می‌نویسیم که اعدادی را از ورودی دریافت می‌کند، و در نهایت میانگین آن‌ها را به کاربر نمایش می‌دهد:

// محل نگهداری اعداد دریافتی از کاربر
var numbers = [];

// عددی که هربار از کاربر در ورودی دریافت می‌شود
var inputNumber = -1;

// مادامی که کاربر عدد 0 را وارد نکند، برنامه به دریافت اعداد ادامه می‌دهد
while(inputNumber !=0){
    inputNumber = parseInt(prompt("یک عدد وارد کنید، برای اتمام دریافت عدد 0 را وارد کنید."));

    // اضافه کردن عدد به آرایه 
    numbers.push(inputNumber);
}

// مجموع اعداد
var sum = 0;

// حساب کردن مجموع اعداد
for(var i=0 ; i < numbers.length ; i++)
    sum = sum + numbers[i];

// محاسبه میانگین
var avg = sum / numbers.length;

alert(avg);
